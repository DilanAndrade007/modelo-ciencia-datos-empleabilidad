C√≥digo 
# Crear mapa de calor usando los datos de habilidades por carrera
# Necesitamos calcular el porcentaje correcto: frecuencia_habilidad / total_frecuencias_carrera

# Primero crear tabla pivot con las frecuencias
pivot_frecuencias = habilidades_carrera.pivot_table(
    index='carrera', 
    columns='habilidad', 
    values='frecuencia', 
    fill_value=0
)

# Calcular porcentajes correctos: cada habilidad / total habilidades por carrera
# Para que la suma de cada fila sea 100%
pivot_habilidades = pivot_frecuencias.div(pivot_frecuencias.sum(axis=1), axis=0) * 100

# Obtener las top 10 carreras m√°s demandadas (sin TOTAL_GENERAL)
top_10_carreras_nombres = tabla_carreras_filtrada.nlargest(10, 'TOTAL_CARRERA')['carrera'].tolist()

# Normalizar nombres de carreras para el matching (reemplazar espacios y may√∫sculas)
carreras_normalizadas = {}
for carrera in pivot_habilidades.index:
    carrera_limpia = carrera.replace('_', ' ').title()
    carreras_normalizadas[carrera] = carrera_limpia

# Encontrar carreras que coincidan (m√°s flexible)
carreras_disponibles = []
for carrera_top in top_10_carreras_nombres:
    for carrera_pivot, carrera_display in carreras_normalizadas.items():
        # Buscar coincidencias flexibles
        if (carrera_top.lower().replace(' ', '_') in carrera_pivot.lower() or 
            carrera_pivot.lower().replace('_', ' ') in carrera_top.lower()):
            carreras_disponibles.append(carrera_pivot)
            break

# Si no hay coincidencias, usar las carreras con m√°s datos
if not carreras_disponibles:
    carreras_disponibles = pivot_habilidades.sum(axis=1).nlargest(8).index.tolist()

# Tomar las top 7 habilidades m√°s frecuentes
top_7_habilidades = pivot_habilidades.sum().nlargest(7).index.tolist()

# Preparar datos para el heatmap (m√°ximo 8 carreras)
heatmap_data = pivot_habilidades.loc[carreras_disponibles[:8], top_7_habilidades]

# Limpiar nombres de carreras para mostrar
carreras_display = [carrera.replace('_', ' ').title() for carrera in heatmap_data.index]
habilidades_display = [hab.title() for hab in heatmap_data.columns]

# Obtener ranking global de habilidades (de m√°s a menos frecuente)
ranking_habilidades = pivot_frecuencias.sum().sort_values(ascending=False)

# Crear mapa de calor usando directamente los valores de porcentaje
# pero con colores ordenados por ranking de frecuencia global
fig = px.imshow(
    heatmap_data.values,
    x=habilidades_display,
    y=carreras_display,
    color_continuous_scale='YlOrRd',
    aspect='auto',
    title=' Mapa de Calor: Distribuci√≥n de Habilidades por Carrera',
    labels={'color': 'Porcentaje (%)'},
    zmin=0,
    zmax=heatmap_data.values.max()
)

# Agregar valores como texto en cada celda
max_valor = heatmap_data.values.max()
for i, carrera in enumerate(carreras_display):
    for j, habilidad in enumerate(habilidades_display):
        valor = heatmap_data.iloc[i, j]
        # Ajustar color del texto basado en el valor relativo
        color_texto = 'white' if valor > (max_valor * 0.5) else 'black'
        fig.add_annotation(
            x=j, y=i,
            text=f'{valor:.1f}%',
            showarrow=False,
            font=dict(color=color_texto, size=9, family='Arial Black')
        )

fig.update_layout(
    height=600,
    xaxis_title="Habilidades Blandas",
    yaxis_title="Carreras",
    xaxis={'tickangle': 45},
    font=dict(size=10)
)

)

# Para que la suma de cada fila sea 100%
pivot_habilidades = pivot_frecuencias.div(pivot_frecuencias.sum(axis=1), axis=0) * 100

# Obtener las top 10 carreras m√°s demandadas (sin TOTAL_GENERAL)
top_10_carreras_nombres = tabla_carreras_filtrada.nlargest(10, 'TOTAL_CARRERA')['carrera'].tolist()

# Normalizar nombres de carreras para el matching (reemplazar espacios y may√∫sculas)
carreras_normalizadas = {}
for carrera in pivot_habilidades.index:
    carrera_limpia = carrera.replace('_', ' ').title()
    carreras_normalizadas[carrera] = carrera_limpia

# Encontrar carreras que coincidan (m√°s flexible)
carreras_disponibles = []
for carrera_top in top_10_carreras_nombres:
    for carrera_pivot, carrera_display in carreras_normalizadas.items():
        # Buscar coincidencias flexibles
        if (carrera_top.lower().replace(' ', '_') in carrera_pivot.lower() or 
            carrera_pivot.lower().replace('_', ' ') in carrera_top.lower()):
            carreras_disponibles.append(carrera_pivot)
            break

# Si no hay coincidencias, usar las carreras con m√°s datos
if not carreras_disponibles:
    carreras_disponibles = pivot_habilidades.sum(axis=1).nlargest(8).index.tolist()

habilidades_display = [hab.title() for hab in heatmap_data.columns]

ranking_habilidades = pivot_frecuencias.sum().sort_values(ascending=False)
